Step 2: MaterialPricingService

Create file: apps/api/src/services/MaterialPricingService.ts

2a. bestAvailableCost()

Resolves the cost basis for a material. Checks in order:

  1. DistributorPrice.effectivePrice (preferred distributor) — only available after MAT-0/MAT-9
  2. MaterialAttributePrice.value (dimension-based catalog price) — available today
  3. null — no cost data available (user must enter manually)
import prisma from "../models/_prismaClient";
import { Decimal } from "@prisma/client/runtime/library";

export async function bestAvailableCost(
    materialId: string
): Promise {
    // Phase 1: Use MaterialAttributePrice (existing system)
    const attrPrice = await prisma.materialAttributePrice.findFirst({
        where: { materialId, isDeleted: false, isArchived: false },
        orderBy: { updatedAt: "desc" }
    });
    if (attrPrice?.value) return attrPrice.value;

    return null;
}
Future upgrade (MAT-0): When DistributorPrice is available, insert a check before MaterialAttributePrice: query DistributorPrice where isPreferred = true for this material’s ManufacturerProduct. The rest of the cascade is unchanged.

2b. resolveMaterialSellPrice()

interface MaterialPriceResolution {
    costPrice: Decimal;
    sellPrice: Decimal;
    marginDollar: Decimal;
    marginPercent: Decimal;
    source: "job" | "material-fixed" | "material-percent" | "group" | "cost";
}

export async function resolveMaterialSellPrice(
    materialId: string,
    jobId?: string | null
): Promise {
    const material = await prisma.material.findUnique({
        where: { id: materialId },
        include: { MaterialGroup: true }
    });
    if (!material) return null;

    const costPrice = await bestAvailableCost(materialId);
    if (!costPrice) return null;

    const group = material.MaterialGroup;
    let sellPrice = costPrice;
    let source: MaterialPriceResolution["source"] = "cost";

    // Tier 3: Job override
    if (jobId) {
        const job = await prisma.job.findUnique({
            where: { id: jobId },
            select: { materialMarkupPercent: true }
        });
        if (job?.materialMarkupPercent != null) {
            sellPrice = costPrice.mul(
                new Decimal(1).plus(job.materialMarkupPercent.div(100))
            );
            source = "job";
            return buildResult(costPrice, sellPrice, source);
        }
    }

    // Tier 2a: Material fixed sell price
    if (material.sellPriceOverride != null) {
        return buildResult(costPrice, material.sellPriceOverride, "material-fixed");
    }

    // Tier 2b: Material markup %
    if (material.markupPercentOverride != null) {
        sellPrice = costPrice.mul(
            new Decimal(1).plus(material.markupPercentOverride.div(100))
        );
        return buildResult(costPrice, sellPrice, "material-percent");
    }

    // Tier 1: Group default
    if (group?.defaultMarkupPercent != null) {
        sellPrice = costPrice.mul(
            new Decimal(1).plus(group.defaultMarkupPercent.div(100))
        );
        return buildResult(costPrice, sellPrice, "group");
    }

    // Fallback: cost pass-through
    return buildResult(costPrice, costPrice, "cost");
}

function buildResult(
    costPrice: Decimal,
    sellPrice: Decimal,
    source: MaterialPriceResolution["source"]
): MaterialPriceResolution {
    const marginDollar = sellPrice.minus(costPrice);
    const marginPercent = costPrice.gt(0)
        ? marginDollar.div(sellPrice).mul(100)
        : new Decimal(0);
    return { costPrice, sellPrice, marginDollar, marginPercent, source };
}
