LEM-3a: Material Billable Rates — Schema + Service (Backend Plan)

Backend half of LEM-3. After this lands:

Estimate: ~1 day. Pure backend — no React. Migration, service, GraphQL resolvers, plus a single set of DocumentAttributeAssignment + DocumentModelFormElements rows for the Job materialMarkupPercent column.

import { prisma } from "../models/_prismaClient";
import { Decimal } from "@prisma/client/runtime/library";

export type MaterialPricingSource =
    | "job"
    | "material-fixed"
    | "material-percent"
    | "group"
    | "cost";

export interface MaterialPriceResolution {
    costPrice: Decimal;
    sellPrice: Decimal;
    marginDollar: Decimal;
    marginPercent: Decimal;
    source: MaterialPricingSource;
}

/**
 * Resolves the cost basis for a material.
 * Order: DistributorPrice (MAT-0, not yet built) -> MaterialAttributePrice -> null.
 */
export async function bestAvailableCost(
    materialId: string
): Promise<Decimal | null> {
    // TODO(MAT-0): When DistributorPrice is available, query for isPreferred=true
    // for this material's ManufacturerProduct first; fall through to MaterialAttributePrice.

    const attrPrice = await prisma.materialAttributePrice.findFirst({
        where: { materialId, isDeleted: false, isArchived: false },
        orderBy: { updatedAt: "desc" }
    });
    return attrPrice?.value ?? null;
}

/**
 * Three-tier material sell-price cascade:
 *   1. Job override (Job.materialMarkupPercent)
 *   2. Material override (sellPriceOverride OR markupPercentOverride)
 *   3. MaterialGroup default (defaultMarkupPercent)
 *   4. Pass-through at cost
 *
 * Returns null only when no cost basis is resolvable.
 */
export async function resolveMaterialSellPrice(
    materialId: string,
    jobId?: string | null
): Promise<MaterialPriceResolution | null> {
    const material = await prisma.material.findUnique({
        where: { id: materialId },
        include: { MaterialGroup: true }
    });
    if (!material) return null;

    const costPrice = await bestAvailableCost(materialId);
    if (!costPrice) return null;

    // Tier 3: Job override
    if (jobId) {
        const job = await prisma.job.findUnique({
            where: { id: jobId },
            select: { materialMarkupPercent: true }
        });
        if (job?.materialMarkupPercent != null) {
            const sell = costPrice.mul(
                new Decimal(1).plus(job.materialMarkupPercent.div(100))
            );
            return buildResult(costPrice, sell, "job");
        }
    }

    // Tier 2a: Material fixed sell price
    if (material.sellPriceOverride != null) {
        return buildResult(costPrice, material.sellPriceOverride, "material-fixed");
    }

    // Tier 2b: Material markup %
    if (material.markupPercentOverride != null) {
        const sell = costPrice.mul(
            new Decimal(1).plus(material.markupPercentOverride.div(100))
        );
        return buildResult(costPrice, sell, "material-percent");
    }

    // Tier 1: Group default
    if (material.MaterialGroup?.defaultMarkupPercent != null) {
        const sell = costPrice.mul(
            new Decimal(1).plus(material.MaterialGroup.defaultMarkupPercent.div(100))
        );
        return buildResult(costPrice, sell, "group");
    }

    // Fallback: pass-through at cost
    return buildResult(costPrice, costPrice, "cost");
}

function buildResult(
    costPrice: Decimal,
    sellPrice: Decimal,
    source: MaterialPricingSource
): MaterialPriceResolution {
    const marginDollar = sellPrice.minus(costPrice);
    const marginPercent = sellPrice.gt(0)
        ? marginDollar.div(sellPrice).mul(100)
        : new Decimal(0);
    return { costPrice, sellPrice, marginDollar, marginPercent, source };
}
Margin convention: margin% is computed against sell price, not cost. This matches Shopify's "Margin" display, the convention referenced in the spec, and is what the customer-facing UI in LEM-3b expects.

import {
    bestAvailableCost,
    resolveMaterialSellPrice
} from "../../services/MaterialPricingService";

// ... in Resolvers: ...
sellPrice: async (parent: { id: string }, args: { jobId?: string | null }) => {
    const r = await resolveMaterialSellPrice(parent.id, args.jobId ?? null);
    return r?.sellPrice?.toNumber() ?? null;
},
marginPercent: async (parent: { id: string }, args: { jobId?: string | null }) => {
    const r = await resolveMaterialSellPrice(parent.id, args.jobId ?? null);
    return r?.marginPercent?.toNumber() ?? null;

costBasis: async (parent: { id: string }) => {
    const cost = await bestAvailableCost(parent.id);
    return cost?.toNumber() ?? null;
},
pricingSource: async (parent: { id: string }, args: { jobId?: string | null }) => {
    const r = await resolveMaterialSellPrice(parent.id, args.jobId ?? null);
    return r?.source ?? "cost";
}

Add a top-level resolveMaterialSellPrice query for callers that don't have a Material loaded:

// In typeDef, alongside `extend type Query`:
type MaterialPriceResolution {
    costPrice: Float!
    sellPrice: Float!
    marginDollar: Float!
    marginPercent: Float!
    source: String!
}

extend type Query {
    materials(options: GetListOptions): GetListResult
    material(id: ID!): Material
    getMaterialRelatedDeleteCount(ids: [ID!]!): JSON
    resolveMaterialSellPrice(materialId: String!, jobId: String): MaterialPriceResolution
}

// In Query: ...
resolveMaterialSellPrice: async (_, { materialId, jobId }) => {
    const r = await resolveMaterialSellPrice(materialId, jobId);
    if (!r) return null;
    return {
        costPrice: r.costPrice.toNumber(),
        sellPrice: r.sellPrice.toNumber(),
        marginDollar: r.marginDollar.toNumber(),
        marginPercent: r.marginPercent.toNumber(),
        source: r.source
    };
}
